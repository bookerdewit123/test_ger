####################### Bid Message ##############################
script WsfMessage GetBidMessage(int inBidID, WsfPlatform inOriginator,bidMessageStruct inBidMessageStruct,WsfPlatform inThreat, fireSolutionStruct inFireSolution)
   //initialize variables
   WsfMessage bidMessage = WsfMessage();

   bidMessage.SetType("BidMessage");
   bidMessage.SetAuxData("BidID", inBidID);
   bidMessage.SetAuxData("BidOriginator", inOriginator);
   bidMessage.SetAuxData("Bid", inBidMessageStruct);
   bidMessage.SetAuxData("FireSolution", inFireSolution);
   bidMessage.SetAuxData("Threat", inThreat); 

   return bidMessage;    
end_script
